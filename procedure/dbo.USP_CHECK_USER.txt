ALTER PROCEDURE [dbo].[USP_CHECK_USER]
(
    @id nvarchar(50),   
    @pw nvarchar(50) 
)  
AS   
 DECLARE @Count int
 SELECT @Count = count(*) FROM F_User WHERE @id = f_id AND @pw = f_pw
 
 if @Count = 1
	Begin
		UPDATE f_User SET f_last_login_date = GETDATE() WHERE f_id = @id
		SELECT 'SUCCESS_LOGIN'
	End
 ELSE
	SELECT 'FAIL_LOGIN'
