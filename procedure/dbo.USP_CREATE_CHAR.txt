ALTER procedure [dbo].[USP_CREATE_CHAR]
	@id nvarchar(50),
	@nick nvarchar(50)
AS
 DECLARE @userIdx int
 SELECT @userIdx = f_idx FROM F_User WHERE f_id = @id
 if @userIdx >= 0
	Begin
		INSERT INTO F_Char (f_nick, f_level, f_exp, f_user_idx) VALUES (@nick, 1, 0, @userIdx)
		SELECT 'REGISTER_SUCCESS'
	End
else
	SELECT 'WRONG_ID'
