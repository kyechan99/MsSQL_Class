ALTER procedure [dbo].[USP_GET_CHAR]
	@id nvarchar(50)
AS
	DECLARE @Count int
	SELECT @Count = count(*) FROM F_User WHERE @id = f_id
 
	if @Count = 1
		Begin
			SELECT * FROM F_Char WHERE f_user_idx = (SELECT f_idx FROM F_User WHERE @id = f_id)
		End
	ELSE
		SELECT 'FAIL_LOGIN'
